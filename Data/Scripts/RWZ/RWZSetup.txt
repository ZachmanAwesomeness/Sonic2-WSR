private function RWZSetup_SpeedUpMusic
	CheckEqual(object[SLOT_MUSICEVENT_CHANGE].type, TypeName[Music Event])
	temp0 = checkResult
	CheckEqual(object[SLOT_MUSICEVENT_CHANGE].propertyValue, MUSICEVENT_TRANSITION)
	temp0 &= checkResult
	CheckEqual(stage.musicFlag, MUSICEVENT_FLAG_NOCHANGE)
	temp0 &= checkResult
	if temp0 == 0
		switch music.currentTrack
		case TRACK_STAGE
			SetMusicTrack("Invincibility_F.ogg", TRACK_INVINCIBLE, MUSIC_LOOP_INV_F)
			SwapMusicTrack("RockWorld_F.ogg", TRACK_STAGE, MUSIC_LOOP_RWZ_F, 8000)
			break

		case TRACK_INVINCIBLE
			SetMusicTrack("RockWorld_F.ogg", TRACK_STAGE, MUSIC_LOOP_RWZ_F)
			SwapMusicTrack("Invincibility_F.ogg", TRACK_INVINCIBLE, MUSIC_LOOP_INV_F, 8000)
			break

		case TRACK_BOSS
		case TRACK_DROWNING
		case TRACK_SUPER
			SetMusicTrack("RockWorld_F.ogg", TRACK_STAGE, MUSIC_LOOP_RWZ_F)
			SetMusicTrack("Invincibility_F.ogg", TRACK_INVINCIBLE, MUSIC_LOOP_INV_F)
			break

		end switch
	else
		stage.musicFlag = MUSICEVENT_FLAG_SPEEDUP
	end if
end function

private function RWZSetup_SlowDownMusic
	CheckEqual(object[SLOT_MUSICEVENT_CHANGE].type, TypeName[Music Event])
	temp0 = checkResult
	CheckEqual(object[SLOT_MUSICEVENT_CHANGE].propertyValue, MUSICEVENT_TRANSITION)
	temp0 &= checkResult
	CheckEqual(stage.musicFlag, MUSICEVENT_FLAG_NOCHANGE)
	temp0 &= checkResult
	if temp0 == 0
		switch music.currentTrack
		case TRACK_STAGE
			SetMusicTrack("Invincibility.ogg", TRACK_INVINCIBLE, MUSIC_LOOP_INV)
			SwapMusicTrack("RockWorld.ogg", TRACK_STAGE, MUSIC_LOOP_RWZ, 12500)
			break

		case TRACK_INVINCIBLE
			SetMusicTrack("RockWorld.ogg", TRACK_STAGE, MUSIC_LOOP_RWZ)
			SwapMusicTrack("Invincibility.ogg", TRACK_INVINCIBLE, MUSIC_LOOP_INV, 12500)
			break

		case TRACK_BOSS
		case TRACK_DROWNING
		case TRACK_SUPER
			SetMusicTrack("RockWorld.ogg", TRACK_STAGE, MUSIC_LOOP_RWZ)
			SetMusicTrack("Invincibility.ogg", TRACK_INVINCIBLE, MUSIC_LOOP_INV)
			break

		end switch
	else
		stage.musicFlag = MUSICEVENT_FLAG_SLOWDOWN
	end if
end function

event ObjectUpdate
	object.PaletteTimer++
		if object.PaletteTimer == 6
			object.PaletteTimer = 0
			RotatePalette(0, 178, 181, false)
		end if

	// Piston Ani Tiles
object.TileTimer++
	GetTableValue(temp0, object.TileFrame, RWZSetup_AniTileDelay)
	if object.TileTimer >= temp0
		object.TileTimer = 0 
		object.TileFrame++
		object.TileFrame %= 6

		temp2 = object.TileFrame
		temp2 += 538
		Copy16x16Tile(73, temp2)
			
		temp2 = 6
		temp2 += object.TileFrame
		temp2 += 538
		Copy16x16Tile(77, temp2)

		temp2 = 12
		temp2 += object.TileFrame
		temp2 += 538
		Copy16x16Tile(78, temp2)

		temp2 = 18
		temp2 += object.TileFrame
		temp2 += 538
		Copy16x16Tile(79, temp2)
	end if

	Object.DeformTimer++
		if Object.DeformTimer >= 2
			Object.DeformTimer = 0
			TileLayer[1].DeformationOffset++
		end if
end event

event ObjectStartup
	SetMusicTrack("RockWorld.ogg", 0, MUSIC_LOOP_RWZ)
	SpeedUpMusic = RWZSetup_SpeedUpMusic
	SlowDownMusic = RWZSetup_SlowDownMusic

	animalType1 = TypeName[Flicky]
	animalType2 = TypeName[Ricky]
	object[SLOT_ZONESETUP].type = TypeName[RWZ Setup]
	object[SLOT_ZONESETUP].priority = PRIORITY_ACTIVE

	SetLayerDeformation(DEFORM_BG, 24, 1, 0, 0, 0)

end event
